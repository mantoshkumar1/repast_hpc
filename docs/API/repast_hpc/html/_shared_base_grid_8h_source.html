200 /**
201  * Grid / Space implementation specialized for the distributed context.
202  * Each SharedBaseGrid of the same name running on different processes
203  * is part of a pan process grid. This class manages this local part of
204  * the grid and its communication with its process neighbors. Users
205  * can specify a buffer size that determines how much of the neighboring grids
206  * are visible in this grid. For example, if this grid originates at 0x0 and ends
207  * at 3x3, a buffer of 1 would make the locations (4,0), (4,1) (4,2) ... (4,4)
208  * and (0,4), (1,4)... (4,4) visible in this grid. The SharedBaseGrid takes many
209  * template parameters. Default variations of these that define typical grids and
210  * spaces are given in SharedGrids in SharedSpace.h
211  *
212  * @tparam T the type of objects contained by this BaseGrid
213  * @tparam GPTransformer transforms cell points according to the topology (e.g. periodic)
214  * of the BaseGrid.
215  * @tparam Adder determines how objects are added to the grid from its associated context.

339 template<typename T, typename GPTransformer, typename Adder, typename GPType>
340 GridDimensions SharedBaseGrid<T, GPTransformer, Adder, GPType>::createSendBufferBounds(Neighbors::Location location) {
341  Point<double> localOrigin = localBounds.origin();
342  Point<double> localExtent = localBounds.extents();
343 
344  switch (location) {
345  double xStart, yStart;
346 case Neighbors::E:
347  xStart = localOrigin.getX() + localExtent.getX() - _buffer;
348  return GridDimensions(Point<double> (xStart, localOrigin.getY()), Point<double> (_buffer, localExtent.getY()));
349 
350 case Neighbors::W:
351  return GridDimensions(localOrigin, Point<double> (_buffer, localExtent.getY()));
352 
353 case Neighbors::N:
354  return GridDimensions(localOrigin, Point<double> (localExtent.getX(), _buffer));
355 
356 case Neighbors::S:
357  yStart = localOrigin.getY() + localExtent.getY() - _buffer;
358  return GridDimensions(Point<double> (localOrigin.getX(), yStart), Point<double> (localExtent.getX(), _buffer));
359 
360 case Neighbors::NE:
361  xStart = localOrigin.getX() + localExtent.getX() - _buffer;
362  return GridDimensions(Point<double> (xStart, localOrigin.getY()), Point<double> (_buffer, _buffer));
363 
364 case Neighbors::NW:
365  return GridDimensions(Point<double> (localOrigin.getX(), localOrigin.getY()), Point<double> (_buffer, _buffer));
366 
367 case Neighbors::SE:
368  xStart = localOrigin.getX() + localExtent.getX() - _buffer;
369  yStart = localOrigin.getY() + localExtent.getY() - _buffer;
370  return GridDimensions(Point<double> (xStart, yStart), Point<double> (_buffer, _buffer));
371 
372 case Neighbors::SW:
373  yStart = localOrigin.getY() + localExtent.getY() - _buffer;
374  return GridDimensions(Point<double> (localOrigin.getX(), yStart), Point<double> (_buffer, _buffer));
375  }
376 
377  return GridDimensions();
378 }

381 template<typename T, typename GPTransformer, typename Adder, typename GPType>
382 void SharedBaseGrid<T, GPTransformer, Adder, GPType>::balance() {
383  int r = comm->rank();
384  typename GridBaseType::LocationMapConstIter iterEnd = GridBaseType::locationsEnd();
385  for (typename GridBaseType::LocationMapConstIter iter = GridBaseType::locationsBegin(); iter != iterEnd; ++iter) {
386  AgentId id = iter->second->ptr->getId();
387  if(id.currentRank() == r){ // Local agents only
388  Point<GPType> loc = iter->second->point;
389  if(!localBounds.contains(loc)){ // If inside bounds, ignore
390  Neighbor* ngh = nghs.findNeighbor(loc.coords());
391  RepastProcess::instance()->moveAgent(id, ngh->rank());
392  }
393  }
394  }

415 template<typename T, typename GPTransformer, typename Adder, typename GPType>
416 void SharedBaseGrid<T, GPTransformer, Adder, GPType>::getAgentsToPush(std::set<AgentId>& agentsToTest, std::map<int, std::set<AgentId> >& agentsToPush){
417 
418  if(_buffer == 0) return; // A buffer zone of zero means that no agents will be pushed.
419 
420  // In a general case, we might not want to do this, but
421  // for the current configuration, we can make this (perhaps much) more efficient
422  // this way:
423  Point<double> localOrigin = localBounds.origin();
424  Point<double> localExtent = localBounds.extents();
425 
426  double xStart = localOrigin.getX() + _buffer;
427  double xEnd = localExtent.getX() - _buffer * 2;
428  double yStart = localOrigin.getY() + _buffer;
429  double yEnd = localExtent.getY() - _buffer * 2;
430 
431  GridDimensions unbuffered(Point<double> (xStart, yStart), Point<double> (xEnd, yEnd));
432 
433  Neighbor* neighbor;
434 
435  neighbor = nghs.neighbor(Neighbors::NW);
436  int NW_rank = (neighbor == 0 ? -1 : neighbor->rank());
437  neighbor = nghs.neighbor(Neighbors::N );
438  int N_rank = (neighbor == 0 ? -1 : neighbor->rank());
439  neighbor = nghs.neighbor(Neighbors::NE);
440  int NE_rank = (neighbor == 0 ? -1 : neighbor->rank());
441  neighbor = nghs.neighbor(Neighbors::E );
442  int E_rank = (neighbor == 0 ? -1 : neighbor->rank());
443  neighbor = nghs.neighbor(Neighbors::SE);
444  int SE_rank = (neighbor == 0 ? -1 : neighbor->rank());
445  neighbor = nghs.neighbor(Neighbors::S );
446  int S_rank = (neighbor == 0 ? -1 : neighbor->rank());
447  neighbor = nghs.neighbor(Neighbors::SW);
448  int SW_rank = (neighbor == 0 ? -1 : neighbor->rank());
449  neighbor = nghs.neighbor(Neighbors::W );
450  int W_rank = (neighbor == 0 ? -1 : neighbor->rank());
451 
452  GridDimensions empty(Point<double>(0,0), Point<double>(0,0));
453 
454  GridDimensions N_bounds = (N_rank > - 1 ? createSendBufferBounds(Neighbors::N) : empty);
455  GridDimensions E_bounds = (E_rank > - 1 ? createSendBufferBounds(Neighbors::E) : empty);
456  GridDimensions S_bounds = (S_rank > - 1 ? createSendBufferBounds(Neighbors::S) : empty);
457  GridDimensions W_bounds = (W_rank > - 1 ? createSendBufferBounds(Neighbors::W) : empty);
458 
459  std::set<AgentId> NW_set;
460  std::set<AgentId> N_set;
461  std::set<AgentId> NE_set;
462  std::set<AgentId> E_set;
463  std::set<AgentId> SE_set;
464  std::set<AgentId> S_set;
465  std::set<AgentId> SW_set;
466  std::set<AgentId> W_set;
467 
468 
469  // Local agents that are in other processes' 'buffer zones' must be exported to those other processes.
470  int r = comm->rank();
471  std::set<AgentId>::iterator idIter = agentsToTest.begin();
472  while(idIter != agentsToTest.end()){
473  AgentId id = *idIter;
474  bool found = false;
475  if(id.currentRank() == r){ // Local agents only
476  std::vector<GPType> locationVector;
477  GridBaseType::getLocation(id, locationVector);
478  Point<GPType> loc(locationVector);
479  if(!unbuffered.contains(loc)){
480  if(W_bounds.contains(loc)){
481  found = true;
482  W_set.insert(id);
483  if(N_bounds.contains(loc)){
484  N_set.insert(id);
485  NW_set.insert(id);
486  }
487  else{
488  if(S_bounds.contains(loc)){
489  S_set.insert(id);
490  SW_set.insert(id);
491  }
492  }
493  }
494  else{
495  if(E_bounds.contains(loc)){
496  found = true;
497  E_set.insert(id);
498  if(N_bounds.contains(loc)){
499  N_set.insert(id);
500  NE_set.insert(id);
501  }
502  else{
503  if(S_bounds.contains(loc)){
504  S_set.insert(id);
505  SE_set.insert(id);
506  }
507  }
508  }
509  else{
510  if(N_bounds.contains(loc)){
511  found = true;
512  N_set.insert(id);
513  }
514  else{
515  if(S_bounds.contains(loc)){
516  found = true;
517  S_set.insert(id);
518  }
519  }
520  }
521  }
522  }
523  }
524  if(found){
525  std::set<AgentId>::iterator tmp = idIter;
526  idIter++;
527  agentsToTest.erase(tmp);
528  }
529  else{
530  idIter++;
531  }
532  }
533 
534  if(NW_set.size() > 0) agentsToPush[NW_rank].insert(NW_set.begin(), NW_set.end());
535  if( N_set.size() > 0) agentsToPush[ N_rank].insert( N_set.begin(), N_set.end());
536  if(NE_set.size() > 0) agentsToPush[NE_rank].insert(NE_set.begin(), NE_set.end());
537  if( E_set.size() > 0) agentsToPush[ E_rank].insert( E_set.begin(), E_set.end());
538  if(SE_set.size() > 0) agentsToPush[SE_rank].insert(SE_set.begin(), SE_set.end());
539  if( S_set.size() > 0) agentsToPush[ S_rank].insert( S_set.begin(), S_set.end());
540  if(SW_set.size() > 0) agentsToPush[SW_rank].insert(SW_set.begin(), SW_set.end());
541  if( W_set.size() > 0) agentsToPush[ W_rank].insert( W_set.begin(), W_set.end());
542
