362 /**
363  * Given an iterator and a number of elements,
364  * creates a data structure that allows efficient access
365  * to those elements. Is only valid as long as the iterator
366  * is valid.
367  *
368  * The basic implementation creates a vector of ordered
369  * pairs linking an integer and an iterator pointing
370  * to an element in the original iteration set. To find
371  * the nth element, the algorithm searches backwards through the
372  * list of 'landmarks', finds the highest landmark lower than
373  * n, chooses the iterator associated with that landmark,
374  * and steps forward until n is reached, adding new landmarks
375  * if appropriate. So given landmarks:
376  *
377  * 0 - pointer to element 0
378  * 100 - pointer to element 100
379  * 200 - pointer to element 200
380  *
381  * if the request for element 438 is given, the algorithm will
382  * search backward and find landmark 200; it will
383  * then step forward, adding landmarks for 300 and 400, until element 438
384  * is reached and returned.
385  *
386  * Assuming that requests are evenly distributed, optimum interval for
387  * landmarks is the square root of the size of the list, and performance
388  * for the algorithm will be in log(size) time.
389  *
390  * Note that other implementations are possible- for example, checking
391  * if enough memory would allow a completely indexed list. A long-term
392  * possibility is allowing the user to specify (for example, specify that
393  * the algorithm with lowest memory cost be used even though memory
394  * is initially available, perhaps because other routines will be filling
395  * that memory while this object is in use).
396  */
397 template<typename I>
398 class RandomAccess{
399 private:
400  I it;
401  I begin;
402 
403  int interval;
404  int maxLandmark;
405  std::vector<std::pair<int, I > > landmarks;
406 
407 public:
408 
409  /**
410  * Constructs a RandomAccess instance for this iterator
411  *
412  * @param beginning
413  * @param size
414  */
415  RandomAccess(I beginning, int size) :
416  interval((int)(sqrt((double)size))), maxLandmark(0), it(beginning), begin(beginning) {
417  landmarks.push_back(std::pair<int, I >(0, beginning));
418  }
419 
420  /**
421  * Gets the element at the specified index
422  *
423  * @param index
424  */
425  I get(int index){
426  bool place = (index > (maxLandmark + interval));
427  typename std::vector<std::pair<int, I > >::iterator lm = landmarks.end();
428  while((--lm)->first > index);
429  int c = lm->first;
430  it = lm->second;
431  if(place){
432  while(c != index){
433  c++;
434  it++;
435  if(c % interval == 0){
436  landmarks.push_back(std::pair<int, I >(c, it));
437  maxLandmark = c;
438  }
439  }
440  }
441  else{
442  while(c != index){
443  c++;
444  it++;
445  }
446  }
447  return it;
448  }
449 
450 };

453 /**
454  * Gets N elements selected randomly from the elements that are indexed by the iterator passed.
455  * Selection is 'without replacement'; no element will appear in the result set twice.
456  * The elements are placed in a set; order is not randomized.
457  *
458  * If the 'selectedElements' set is not empty initially, the elements chosen will be added
459  * to it with duplication prevented- in other words, the elements initially in the
460  * set will not be selected from the population and added to the set by this routine.
461  *
462  * If it is not possible to select all of the agents requested
463  * because the number requested exceeds the number of unique agents that can be added
464  * to the result set, then all of the agents from the iterator will be added to the set.
465  *
466  * If the optional 'remove' parameter is set to true, the elements in the
467  * original 'selectedElements' set will be removed before the set is
468  * returned.
469  *
470  * @param iteratorStart the start of the iterated list of elements from which
471  * selection will be made
472  * @param size the size of the source population- i.e., the number of elements
473  * in the list pointed to by iteratorStart
474  * @param count number of elements to be selected
475  * @param [out] selectedElements collection into which selected elements
476  * are placed. If not empty, elements in initial set are excluded from
477  * selection.
478  * @param remove if true, the elements in the original set will be removed from
479  * the set returned.
480  *
481  * @tparam T the type of the elements to be selected (i.e. the agent type)
482  * @tparam I the type of the iterator to be used to pass through the population
483  * to be sampled
484  */
485 template<typename T, typename I>
486 void selectNElementsAtRandom(I iterator, int size, unsigned int count, std::set<T*>&selectedElements, bool remove = false){
487  // When 'selectedElements' is large in comparison to size, or 'count' is
488  // large in comparison to size, or both, a potential problem arises; this is both
489  // a performance issue and a potential error (infinite loop). The number
490  // of valid selections remaining in the original population will be
491  // the size of the population minus the number of members of the original
492  // population already in the set of 'selectedElements'. There is a performance
493  // problem that can arise if the number of valid selections approaches
494  // zero, and an infinite loop arises if that number reaches zero. However, a
495  // complication is that we do not assume that all members in
496  // 'selectedElements' are drawn from the same population that 'iterator' describes.
497 
498  // At a general level, the solution to this is that when the number of valid
499  // selections is expected to become small, the algorithm will switch to
500  // selecting the elements that will _not_ be added to the final set, rather
501  // than those that will.
502 
503  // However, the more immediate issue is determining whether the problem
504  // will arise at all, and a key question is how many members of 'selectedElements'
505  // are drawn from the population. This, however, is costly to check, so
506  // it should only be checked if it seems possible to lead to a problem.
507 
508  // A ceiling; the actual value may be lower
509  int maxAlreadySelected = selectedElements.size();
510 
511  // We are not certain _any_ elements in the selectedElements set are from this population
512  int knownSelectedElementsFromPop = 0;
513 
514  // This is a floor; there are at least this many available, but possibly more
515  int minAvailable = size - maxAlreadySelected;
516 
517  // If you are requesting more than might be available, you must determine
518  // the actual number available.
519  if(count > minAvailable){
520  if(maxAlreadySelected > 0){
521  I tempIt = iterator;
522  for(int i = 0; i < size; i++){
523  T* ptr = (&**tempIt);
524  typename std::set<T*>::iterator found = selectedElements.find(ptr);
525  if(found != selectedElements.end()) knownSelectedElementsFromPop++;
526  tempIt++;
527  }
528  maxAlreadySelected = knownSelectedElementsFromPop;
529  minAvailable = size - maxAlreadySelected;
530  }
531  }
532  // If removing the original elements, will need a copy of them
533  typename std::vector<T*> tempToRemove;
534  if(remove){
535  tempToRemove.assign(selectedElements.begin(), selectedElements.end());
536  }
537 
538  // There is no way to satisfy the request; copy all and return
539  if(count > minAvailable){
540  I it = iterator;
541  for(int i = 0; i < size; i++){
542  selectedElements.insert(&**it);
543  it++;
544  }
545  }
546  else{
547  RandomAccess<I> ra(iterator, size);
548  IntUniformGenerator rnd = Random::instance()->createUniIntGenerator(0, size - 1);
549  T* ptr;
550 
551  // If the count of elements is very high in proportion to the population, faster to
552  // choose agents that will _not_ be in the final set and then switch...
553  // First the normal case, in which the number of agents is low
554  if(count <= ((double)minAvailable)*0.6){ // Tests suggest that 2/3 is probably too high, so adjusted to .6; TODO optimize or analytically solve
555  for (unsigned int i = 0; i < count; i++)
556  do { ptr = &**ra.get(rnd.next()); } while(!(selectedElements.insert(ptr).second));
557  }
558  else{ // Now the other case
559  std::set<T*> elementsThatWillNotBeAdded;
560 
561  if(selectedElements.size() > 0){
562  if(selectedElements.size() == knownSelectedElementsFromPop){ // Maybe we already checked and they ALL belong
563  elementsThatWillNotBeAdded.insert(selectedElements.begin(), selectedElements.end());
564  }
565  else{
566  I tempIt = iterator;
567  for(int i = 0; i < size; i++){
568  ptr = (&**tempIt);
569  if(selectedElements.find(ptr) != selectedElements.end()) elementsThatWillNotBeAdded.insert(ptr);
570  tempIt++;
571  }
572  }
573  }
574  // Note: duplicate element will not be inserted; failure will leave size of elementsThatWillNotBeAdded unchanged.
575  while((size - elementsThatWillNotBeAdded.size()) > count) elementsThatWillNotBeAdded.insert(&**ra.get(rnd.next()));
576 
577  // Once the set of elements that will not be added is complete, cycle through the iterator and add
578  // all of those elements that are not in elementsThatWillNotBeAdded
579  I toAdd = iterator;
580  typename std::set<T*>::iterator notFound = elementsThatWillNotBeAdded.end();
581  for(int i = 0; i < size; i++){
582  ptr = &**toAdd;
583  if(elementsThatWillNotBeAdded.find(ptr) == notFound) selectedElements.insert(ptr);
584  toAdd++;
585  }
586  }
587  }
588  // Before returning, remove the elements from the set, if the user requested
589  if(remove){
590  typename std::vector<T*>::iterator toRemove = tempToRemove.begin();
591  while(toRemove != tempToRemove.end()){
592  selectedElements.erase(*toRemove);
593  toRemove++;
594  }
595  }
596 }

660  selectedElements.clear();
661  selectNElementsAtRandom(iterator, size, count, selectedElementSet, remove);
662  shuffleSet(selectedElementSet, selectedElements);
