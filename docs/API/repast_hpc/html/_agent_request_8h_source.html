53 /**
54  * Encapsulates a request made by one process for agents in another.
55  *
56  * Includes a list of requests and a list that represents cancellations
57  * of previous requests.
58  */
59 class AgentRequest {

61  friend std::ostream& operator<<(std::ostream& os, const AgentRequest& request);

150  /**

233  /**
